026/**
027 * Stores a lambda (a triple of bound variables, a body, and the environment it was created in)
028 *
029 * @author Koushik Sen
030 * @author Alex Reinking
031 */
032class Lambda {
033    final String boundVar;
034    final InternalNode body;
035    final Environment env;
036
037    public Lambda(String boundVar, InternalNode body, Environment env) {
038        this.boundVar = boundVar;
039        this.body = body;
040        this.env = env;
041    }
042
043    @Override
044    public String toString() {
045        StringBuilder sb = new StringBuilder();
046        sb.append(env);
047        sb.append(" fun (");
048        if (boundVar != null) {
049            sb.append(boundVar);
050        }
051        sb.append(") ").append(body);
052        return sb.toString();
053    }
054}
